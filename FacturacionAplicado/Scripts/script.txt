create database Facturacion_db;

use facturaciondbapl;

Create Table Usuarios
(
	id int primary key auto_increment,
	Nombre varchar(30),
	Clave varchar(8),
	NombreUsuario varchar(8),
	Fecha varchar(30),
	Comentario varchar(100)


);

insert into Usuarios(Nombre,Clave,NombreUsuario,Fecha,Comentario) values('Albert Lopez','123456','root','2018/6/9','Developer');

Create Table Producto
(
	id int primary key auto_increment,
	Descripcion varchar(400),
	Precio decimal,
	DepartamentoId int,
	Cantidad int

);

create table Departamento
(
	id int primary key auto_increment,
	Nombre varchar(50)

);

create table Cliente
(
	id int primary key auto_increment,
	Nombre varchar(30),
	Direccion varchar(100),
	Cedula varchar(15),
	Telefono varchar(15)

);

create table Factura
(
	id int primary key auto_increment,
	Monto decimal,
	UsuarioId int,
	ClienteId int,
	Fecha varchar(30),
	Descripcion varchar(400),
	FormaDePago varchar(50),
	Devuelta decimal,
	EfectivoRecibido decimal
);

create table FacturaDetalle
(
	id int primary key auto_increment,
	FacturaId int,
	ProductoId int,
	Cantidad int,
	Precio decimal,
	Descripcion varchar(400)
);
